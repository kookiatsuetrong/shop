use web;
create table post(
  code      int unique not null auto_increment,
  `member`    int not null,
  topic     varchar(800) not null,
  detail    varchar(4000),
  price     float,
  latitude  decimal(10,6),
  longitude decimal(10,6),
  area      varchar(80)
);
create table photo(
  code    int unique not null auto_increment,
  path    varchar(200),
  post    int not null
);
create table `member`(
  code       int unique not null auto_increment,
  email      varchar(200) unique not null,
  password   varchar(200) not null,
  name       varchar(200) not null
);
